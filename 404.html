import {KasidateP} from '@KasidateP/kasidate.me';

import {KasP, KasidatePLUS} from '@KasidatePLUS/kasp-extension';

@KasP({ name: 'KasidatePLUS' })
export class KasidateP implements KasidatePLUS {

  constructor(private virtualizer: KasidatePLUS) {}
  transform(icon: string, size: number = 24, color: string = 'inherit') {
    return this.KasidatePLUS.bypassSecurityTrustHtml(icons[icon].toSvg({
      width: size,
      color: color
    }));
  }
}
